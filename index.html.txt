                            the raw energy of urban environments, and the quiet moments of introspection that define our human experience.
                        </p>
                        <p>
                            With a background in architecture and fine arts from RISD, I approach each canvas as a spatial problem— 
                            balancing negative space with explosive color, geometric structure with organic flow. My process involves 
                            building up layers over weeks or months, then selectively revealing or obscuring elements to create depth 
                            and history within the surface.
                        </p>

                            My work has been featured in Architectural Digest, collected by designers across the US and Europe,

        // Cart Functions
        function toggleCart() {
            const sidebar = document.getElementById('cart-sidebar');
            const overlay = document.getElementById('cart-overlay');
            sidebar.classList.toggle('open');
            overlay.classList.toggle('hidden');
            document.body.style.overflow = sidebar.classList.contains('open') ? 'hidden' : '';
        }

        function addToCart(artworkId) {
            const artwork = artworks.find(a => a.id === artworkId);
            if (!artwork || !artwork.available) return;
            
            const existing = cart.find(item => item.id === artworkId);
            if (existing) {
                existing.quantity++;
            } else {
                cart.push({ ...artwork, quantity: 1 });
            }
            
            updateCartUI();
            toggleCart();
        }

        function removeFromCart(artworkId) {
            cart = cart.filter(item => item.id !== artworkId);
            updateCartUI();
        }

        function updateCartUI() {
            const count = cart.reduce((sum, item) => sum + item.quantity, 0);
            const countEl = document.getElementById('cart-count');
            countEl.textContent = count;
            countEl.style.opacity = count > 0 ? '1' : '0';
            
            const itemsContainer = document.getElementById('cart-items');
            const totalEl = document.getElementById('cart-total');
            
            if (cart.length === 0) {
                itemsContainer.innerHTML = `
                    <div class="text-center text-gray-500 py-12">
                        <i data-lucide="shopping-bag" class="w-12 h-12 mx-auto mb-4 opacity-30"></i>
                        <p>Your collection is empty</p>
                    </div>
                `;
                totalEl.textContent = '$0';
            } else {
                itemsContainer.innerHTML = cart.map(item => `
                    <div class="flex gap-4">
                        <img src="${item.image}" class="w-20 h-24 object-cover">
                        <div class="flex-1">
                            <h4 class="font-serif text-lg">${item.title}</h4>
                            <p class="text-sm text-gray-500 mb-2">${item.medium}</p>
                            <div class="flex justify-between items-center">
                                <span class="text-sm">Qty: ${item.quantity}</span>
                                <span class="font-medium">$${(item.price * item.quantity).toLocaleString()}</span>
                            </div>
                        </div>
                        <button onclick="removeFromCart('${item.id}')" class="text-gray-400 hover:text-red-500 transition-colors">
                            <i data-lucide="x" class="w-5 h-5"></i>
                        </button>
                    </div>
                `).join('');
                
                const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
                totalEl.textContent = '$' + total.toLocaleString();
            }
            
            lucide.createIcons();
        }

        function initAnimations() {

                yPercent: 30,
                ease: "none",
                scrollTrigger: {
                    trigger: "#home",
                    start: "top top",
                    end: "bottom top",
                    scrub: true
                }
            });
            
            // Fade up elements

                gsap.fromTo(element, 
                    { opacity: 0, y: 30 },
                    {
                        opacity: 1,
                        y: 0,
                        duration: 1,
                        ease: "power3.out",
                        scrollTrigger: {
                            trigger: element,
                            start: "top 85%",
                            toggleActions: "play none none reverse"
                        }
                    }
                );
            });
